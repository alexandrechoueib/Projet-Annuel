109  void compute() {
110  m_sum = (DataType) 0;
111  for (auto val : m_values) {
112  m_sum += val.value();
113  }
114  m_avg = static_cast<RealType>(m_sum) / m_values.size();
115  Insider insider(m_avg, m_values.size());
116  insider = for_each(m_values.begin(), m_values.end(), insider);
117  m_var = insider.variance();
118  m_sdv = sqrt(m_var);
119 
120  m_min = m_values[0].value();
121  m_max = m_values[m_values.size()-1].value();
122 
123  m_freq_min = 0;
124  for (auto val : m_values) {
125  if (val.value() == m_min) ++m_freq_min;
126  }
127  m_freq_max = 0;
128  for (auto val : m_values) {
129  if (val.value() == m_max) ++m_freq_max;
130  }
131 
132  if ((m_values.size() % 2) == 0) {
133  m_med = static_cast<RealType>(m_values[m_values.size() / 2 - 1].value() +
134  m_values[m_values.size() / 2].value()) / 2.0;
